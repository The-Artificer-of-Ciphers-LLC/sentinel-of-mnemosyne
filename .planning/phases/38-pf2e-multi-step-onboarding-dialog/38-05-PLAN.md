    ```python
    """Pre-router gate for the multi-step onboarding dialog.

    Inserted into discord_router_bridge.route_message BEFORE command_router so
    that plain-text replies inside a draft-bearing thread are consumed as
    onboarding answers, not routed to the AI. On miss, returns None and the
    bridge falls through to command_router unchanged.

    D-01, D-02 from 38-CONTEXT.md.
    """
    from __future__ import annotations
    import logging
    import discord
    import httpx

    logger = logging.getLogger(__name__)

    async def maybe_consume_as_answer(*, user_id: str, message: str, channel,
                                       sentinel_client, http_client) -> str | None:
        # Hit condition 1: message must be non-empty and not a `:` command
        if not message or not message.strip():
            return None
        if message.lstrip().startswith(":"):
            return None
        # Hit condition 2: channel must be a Thread
        if not isinstance(channel, discord.Thread):
            return None
        # Hit condition 3: draft must exist for (thread.id, user_id)
        # Lazy import — keeps this module decoupled from dialog state internals at import time
        import pathfinder_player_dialog as ppd
        try:
            draft = await ppd.load_draft(channel.id, str(user_id), http_client=http_client)
        except httpx.RequestError:
            logger.warning("dialog_router pre-check Obsidian error — falling through to command_router", exc_info=True)
            return None
        except Exception:
            logger.exception("dialog_router pre-check unexpected error — falling through")
            return None
        if draft is None:
            return None
        # Hit — consume as answer
        return await ppd.consume_as_answer(
            thread=channel, user_id=str(user_id), message_text=message,
            sentinel_client=sentinel_client, http_client=http_client,
        )
    ```

    """Bridge wrapper for top-level Discord message routing/subcommands.

    Pre-router gate (Phase 38, D-01): dialog_router.maybe_consume_as_answer
    runs first; on hit (returns str), its response is returned directly.
    On miss (returns None), command_router runs unchanged.
    """
    from __future__ import annotations
    from dialog_router import maybe_consume_as_answer

    async def route_message(*, user_id: str, message: str, attachments, channel,
                            command_router, handle_subcommand, call_core,
                            subcommand_help: str, sentinel_client=None, http_client=None):
        # Pre-router gate — consume thread plain-text as onboarding answers.
        if sentinel_client is not None and http_client is not None:
            consumed = await maybe_consume_as_answer(

            )

                return consumed
        # Fall through — preserve existing command_router contract verbatim.
        return await command_router.route_message(
            user_id=user_id, message=message, attachments=attachments, channel=channel,
            handle_subcommand=handle_subcommand, call_core=call_core,
            subcommand_help=subcommand_help,
        )

    async def handle_subcommand(*, subcmd: str, args: str, user_id: str, attachments, channel,

        )
