```python
import pytest


@pytest.mark.xfail(reason="Wave 1 bot dispatch not yet implemented")
async def test_pf_dispatch_create():
    """`:pf npc create` routes to SentinelCoreClient.post_to_module (bot-layer, NPC-01)."""
    mock_result = {
        "status": "created", "slug": "varek",
        "path": "mnemosyne/pf2e/npcs/varek.md",
        "name": "Varek", "level": 1, "ancestry": "Gnome",
    }
    with patch.object(bot, "_sentinel_client") as mock_client:
        mock_client.post_to_module = AsyncMock(return_value=mock_result)
        result = await bot.handle_sentask_subcommand(
            "pf", "npc create Varek | young gnome rogue", "user-123",
            attachments=None,
        )
    # Result should be a non-empty string (embed text or JSON summary)
    assert isinstance(result, str)
    assert len(result) > 0
    mock_client.post_to_module.assert_called_once()
    call_args = mock_client.post_to_module.call_args
    # First positional arg should be the module path
    assert "npc/create" in call_args[0][0]


@pytest.mark.xfail(reason="Wave 1 bot dispatch not yet implemented")
async def test_pf_dispatch_relate_invalid():
    """`:pf npc relate` with invalid relation type returns error string without calling module."""
    with patch.object(bot, "_sentinel_client") as mock_client:
        mock_client.post_to_module = AsyncMock(return_value={})
        result = await bot.handle_sentask_subcommand(
            "pf", "npc relate Varek enemies-with Baron Aldric", "user-123",
            attachments=None,
        )
    # Should return an error mentioning valid options — module should NOT be called
    assert isinstance(result, str)
    assert "enemies-with" in result.lower() or "valid" in result.lower() or "invalid" in result.lower()
    mock_client.post_to_module.assert_not_called()
```
