```python
async def build_npc_roster_cache(obsidian_client) -> dict:
    """Build {lowercase_name_or_slug: slug} mapping from mnemosyne/pf2e/npcs/*.md (D-22)."""
    try:
        paths = await obsidian_client.list_directory("mnemosyne/pf2e/npcs/")
    except Exception as exc:
        logger.warning("NPC roster cache build failed (Obsidian unreachable?): %s", exc)
        return {}
    result: dict[str, str] = {}
    for path in paths:
        slug = path.removeprefix("mnemosyne/pf2e/npcs/").removesuffix(".md")
        result[slug] = slug  # slug → slug
        try:
            content = await obsidian_client.get_note(path)
            if content:
                # Inline frontmatter parse — reuse pattern from app/routes/npc.py
                import yaml as _yaml
                _raw = content.split("---", 2)
                fm = _yaml.safe_load(_raw[1]) if len(_raw) >= 3 else {}
                name = (fm or {}).get("name", "")
                if name:
                    result[name.lower()] = slug
        except Exception:
            pass
    return result
```
